\renewcommand{\INSTmnemonic}	{mulio}
\renewcommand{\INSTdesc}	{Multiply Immediate with Overflow}
\renewcommand{\INSTopcode}	{0xB2}
\renewcommand{\INSToptype}	{ThreeReg}
\renewcommand{\INSTfieldA}	{des}
\renewcommand{\INSTfieldB}	{src1}
\renewcommand{\INSTfieldC}	{const8}

\renewcommand{\INSTsemantic}	{
	\begin{tabular}{lll}
	\Reg{des} & $\leftarrow$ & (\Reg{src1} $\times$ {\em const8})[31..0] \\
	\Reg{des + 1} & $\leftarrow$ & (\Reg{src1} $\times$ {\em const8})[63..32] \\
	\end{tabular}
}

\renewcommand{\INSTprose}	{

	64-bit multiplication:
	\Reg{des} gets the low-order 32 bits of the result while \Reg{des + 1}
	gets the high-order 32 bits of the result.

	If {\em des} is not even, an illegal instruction exception
	occurs.
}

\renewcommand{\INSTexceptions}	{
		Illegal instruction
	}
