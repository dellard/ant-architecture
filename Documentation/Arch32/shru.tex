\renewcommand{\INSTmnemonic}	{shru}
\renewcommand{\INSTdesc}	{Shift Right Unsigned}
\renewcommand{\INSTopcode}	{0x8D}
\renewcommand{\INSToptype}	{ThreeReg}
\renewcommand{\INSTfieldA}	{des}
\renewcommand{\INSTfieldB}	{src1}
\renewcommand{\INSTfieldC}	{src2}

\renewcommand{\INSTsemantic}	{
	\Reg{des} $\leftarrow$ \Reg{src1} $>>$ \Reg{src2}
}

\renewcommand{\INSTprose} {
	Place into \Reg{des} the value in \Reg{src1} shifted right
	by the number of bits represented by the value of the low five
	bits in \Reg{src2}, filling the high order bits with 0's.
}


\renewcommand{\INSTexceptions}	{}
