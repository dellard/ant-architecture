\renewcommand{\INSTmnemonic}	{jnz}
\renewcommand{\INSTdesc}	{Jump if Not Zero}
\renewcommand{\INSTopcode}	{0xD3}
\renewcommand{\INSToptype}	{ThreeReg}
\renewcommand{\INSTfieldA}	{des}
\renewcommand{\INSTfieldB}	{src1}
\renewcommand{\INSTfieldC}	{src2}

\renewcommand{\INSTsemantic}	{

	If the value in \Reg{src1} is non-zero, then branch to the
	address in \Reg{src2}.  In either case, the address of the
	current instruction is copied into \Reg{des}.

}

\renewcommand{\INSTexceptions}	{An alignment exception occurs if
	the branch address is not divisible by 4.}
