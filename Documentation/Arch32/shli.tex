\renewcommand{\INSTmnemonic}	{shli}
\renewcommand{\INSTdesc}	{Shift Left Immediate}
\renewcommand{\INSTopcode}	{0x9E}
\renewcommand{\INSToptype}	{TwoReg}
\renewcommand{\INSTfieldA}	{des}
\renewcommand{\INSTfieldB}	{src1}
\renewcommand{\INSTfieldC}	{const8}

\renewcommand{\INSTsemantic}	{
	\Reg{des} $\leftarrow$ \Reg{src1} $<<$ {\em const8}
}

\renewcommand{\INSTprose} {
	Place into \Reg{des} the value in \Reg{src1} shifted left
	by the number of bits specified by the value of the low
	five bits of {\em const8}, filling the low order bits with 0's.
}

\renewcommand{\INSTexceptions}	{}
