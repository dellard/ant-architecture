\renewcommand{\INSTmnemonic}	{tlbse}
\renewcommand{\INSTdesc}	{Store TLB Entry}
\renewcommand{\INSTopcode}	{0x42}
\renewcommand{\INSToptype}	{ThreeReg}
\renewcommand{\INSTfieldA}	{\INSTunusedField}
\renewcommand{\INSTfieldB}	{src1}
\renewcommand{\INSTfieldC}	{src2}

\renewcommand{\INSTsemantic}	{
}

\renewcommand{\INSTprose}	{

	Store TLB entry.  Constructs a TLB entry from the integer
	register pair indicated by \Reg{src2} and stores it in the
	TLB at the index specified by \Reg{src1}.

	The lower word of the TLB entry is stored from register
	\Reg{src2}, and the upper word of the TLB entry is stored from
	register \Reg{src2+1},

	If the index is greater than or equal to the number of TLB
	entries, a TLB invalid index exception occurs.

	If {\em des} is not even, an illegal instruction exception
	occurs.
}


\renewcommand{\INSTexceptions}	{
	Privileged instruction,
	Illegal instruction,
	TLB invalid index
	}
