\renewcommand{\INSTmnemonic}	{subio}
\renewcommand{\INSTdesc}	{Subtract Immediate with Overflow}
\renewcommand{\INSTopcode}	{0xB1}
\renewcommand{\INSToptype}	{ThreeReg}
\renewcommand{\INSTfieldA}	{des}
\renewcommand{\INSTfieldB}	{src1}
\renewcommand{\INSTfieldC}	{constr8}

\renewcommand{\INSTsemantic}	{
	\begin{tabular}{lll}
	\Reg{des} & $\leftarrow$ & (\Reg{src1} - {\em const8})[31..0] \\
        \Reg{des + 1} & $\leftarrow$ & -1, if underflow\\
        \Reg{des + 1} & $\leftarrow$ &  0, if no overflow\\
        \Reg{des + 1} & $\leftarrow$ &  1, if overflow\\
	\end{tabular}
}

\renewcommand{\INSTprose}	{

	64-bit subtraction:
        \Reg{des} gets the low-order 32 bits of the result while \Reg{des + 1}
        gets -1,0, or 1.

	If {\em des} is not even, an illegal instruction exception
	occurs.

}

\renewcommand{\INSTexceptions}	{
		Illegal instruction
	}
